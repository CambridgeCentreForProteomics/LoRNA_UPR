Effect of eIF3e KD on membrane localisation

Here, we use Lin & Li et al (2020) (https://pubmed.ncbi.nlm.nih.gov/32589965/) to explore whether eIF3e may have a role in promoting UPR-resistant membrane localisation.

Load libraries

library(tidyverse)
library(camprotR)

Load model outputs from previous notebook

penalised_fits <- readRDS('./out/penalised_fits.rds')
residuals_annot <- readRDS('./out/residuals_annot.rds')

Start with changes in ribosome density in codons 25-75.

to_plot <- residuals_annot$transcript %>%
  mutate(increased_5prime_ribo_density=id %in% lin_li_2020_eif3e_ribo_density_5prime$trans_id)


legend_name <- "eIF3e KD\nIncreased ribosome\ndensity in\ncodons 25-75\nLin & Li et al (2020)"
p <- to_plot %>%
  ggplot(aes(DMSO, Tg, colour=increased_5prime_ribo_density)) +
  geom_point(aes(alpha=increased_5prime_ribo_density, size=increased_5prime_ribo_density)) +
  geom_smooth() +
  theme_camprot(base_size=15, border=FALSE, base_family='sans') +
  scale_colour_manual(values=c('grey20', get_cat_palette(7)[7]), name=legend_name) +
  scale_size_manual(values=c(.2, .5), name=legend_name) +
  scale_alpha_manual(values=c(.2, .5), name=legend_name) +
  theme(strip.background=element_blank()) +
  xlim(0,1) + ylim(0,1) +
  xlab('Membrane proportion Control') +
  ylab('Membrane proportion UPR')

print(p)
ggsave('../../../../../5_manuscript_figures/Figure_5/external/Lin_li_rd.png', height=3.5, width=6)
ggsave('../../../../../5_manuscript_figures/Figure_5/external/Lin_li_rd.pdf', height=3.5, width=6)
